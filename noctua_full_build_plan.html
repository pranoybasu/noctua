Webhook route with signature verification
Node.js
▶
// apps/web/app/api/webhook/github/route.ts import { createHmac, timingSafeEqual } from "crypto" function verifySig(raw: string, sig: string) { const expected = "sha256=" + createHmac("sha256", process.env.GITHUB_WEBHOOK_SECRET!).update(raw).digest("hex") return timingSafeEqual(Buffer.from(sig), Buffer.from(expected)) } export async function POST(req: Request) { const raw = await req.text() const sig = req.headers.get("x-hub-signature-256") ?? "" if (!verifySig(raw, sig)) return Response.json({ error: "Forbidden" }, { status: 403 }) const ev = req.headers.get("x-github-event") const body = JSON.parse(raw) if (ev === "pull_request" && ["opened","synchronize","reopened"].includes(body.action)) { // Push job into Upstash Redis queue await fetch(`${process.env.UPSTASH_REDIS_URL}/lpush/noctua:jobs`, { method: "POST", headers: { Authorization: `Bearer ${process.env.UPSTASH_TOKEN}` }, body: JSON.stringify({ pr_number: body.pull_request.number, repo: body.repository.full_name, diff_url: body.pull_request.diff_url, title: body.pull_request.title, author: body.pull_request.user.login, persona: body.installation?.id // resolved to persona from DB }) }) } return Response.json({ ok: true }) }

PERSONAS = { "strict": { "system": """You are a meticulous senior engineer. Call out EVERY issue. Be direct, cite specific line numbers, no encouragement. Fail low-quality PRs without hesitation.""", "score_weight": 0.6 # harder scorer }, "mentor": { "system": """You are a kind senior engineer teaching a junior dev. Explain WHY every suggestion matters. Use analogies. Always end with one genuine compliment. Be warm and patient.""", "score_weight": 1.1 }, "fast": { "system": """You are a fast-moving startup reviewer. Focus ONLY on blocking issues (security holes, broken logic). Skip style nits. Keep your review under 5 bullet points. Ship fast.""", "score_weight": 0.9 }, "balanced": { "system": "You are an experienced reviewer. Balance thoroughness with speed.", "score_weight": 1.0 } } async def ai_review(diff, title, persona="balanced"): cfg = PERSONAS.get(persona, PERSONAS["balanced"]) resp = await groq.chat.completions.create( model="llama3-70b-8192", messages=[ {"role":"system","content":cfg["system"]}, {"role":"user","content":f"PR: {title}\n\nDiff:\n{diff[:8000]}"} ], response_format={"type":"json_object"} ) import json result = json.loads(resp.choices[0].message.content) result["quality_score"] = int(result.get("quality_score",70) * cfg["score_weight"]) return result

★ Code DNA fingerprinting — Noctua original
UniquePython
▶
OriginalEvery author leaves coding fingerprints — average line length, comment frequency, nesting depth, naming patterns. Noctua tracks these per-author across PRs to build a "Code DNA" profile shown on the leaderboard.
apps/engine/services/dna.py
import ast, re from dataclasses import dataclass, asdict @dataclass class CodeDNA: author: str avg_line_length: float comment_ratio: float # comments / total lines avg_nesting_depth: float # how deep the if/for nesting goes uses_type_hints: bool snake_case_score: float # % of identifiers in snake_case def build_code_dna(code: str, author: str) -> CodeDNA: lines = code.splitlines() total = len(lines) or 1 comments = sum(1 for l in lines if l.strip().startswith("#")) avg_len = sum(len(l) for l in lines) / total depth, max_d = 0, 0 for l in lines: depth += l.count("if ") + l.count("for ") + l.count("while ") depth -= l.count("return") + l.count("break") max_d = max(max_d, depth) identifiers = re.findall(r'\b([a-z_][a-zA-Z0-9_]*)\b', code) snake = sum(1 for i in identifiers if re.fullmatch(r'[a-z_][a-z0-9_]*',i)) hints = bool(re.search(r':\s*(int|str|bool|float|list|dict|Optional)', code)) return CodeDNA(author, round(avg_len,1), round(comments/total,2), round(max_d/total*10,1), hints, round(snake/max(len(identifiers),1),2))

★ Pre-flight check API — Noctua original
UniquePython
▶
OriginalA GitHub Action step that hits Noctua's /preflight endpoint BEFORE the PR is opened. It fails the action if your local changes have critical security issues or extremely low quality — catching problems before review even starts.
apps/engine/routers/preflight.py
from fastapi import APIRouter from services.scanner import scan_code router = APIRouter() @router.post("/preflight") async def preflight(payload: dict): """Called by GitHub Action before PR opens""" code = payload.get("code", "") issues = scan_code(code) critical = [i for i in issues if i["severity"] == "critical"] passed = len(critical) == 0 return { "passed": passed, "critical_count": len(critical), "message": "All clear" if passed else f"Blocked: {len(critical)} critical issue(s) found", "issues": critical }
